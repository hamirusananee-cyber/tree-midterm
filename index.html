#include using namespace std; // นิยามโครงสร้าง Node struct Node { int data; Node *left, *right; Node(int val) : data(val), left(nullptr), right(nullptr) {} }; // ฟังก์ชันเพิ่มข้อมูล (ตามกฎ BST) Node* insert(Node* root, int val) { if (!root) return new Node(val); if (val < root->data) root->left = insert(root->left, val); else root->right = insert(root->right, val); return root; } // ฟังก์ชันแสดงผลแบบ Inorder (เรียงน้อยไปมาก) void inorder(Node* root) { if (!root) return; inorder(root->left); cout << root->data << " "; inorder(root->right); } int main() { Node* root = nullptr; int values[] = {8, 3, 10, 1, 6, 14}; for (int v : values) { root = insert(root, v); } cout << "Inorder traversal: "; inorder(root); // ผลลัพธ์: 1 3 6 8 10 14 return 0; }
